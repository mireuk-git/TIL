## 28279
[link](https://www.acmicpc.net/problem/28279)
### python
```python
from collections import deque

n=int(input())
d=deque()
for i in range(n):
    command=list(map(int,input().split()))
    if command[0]==1:
        d.appendleft(command[1])
    elif command[0]==2:
        d.append(command[1])
    elif command[0]==3:
        if len(d)!=0: print(d.popleft())
        else: print(-1)
    elif command[0]==4:
        if len(d)!=0: print(d.pop())
        else: print(-1)
    elif command[0]==5:
        print(len(d))
    elif command[0]==6:
        if len(d)==0: print(1)
        else: print(0)
    elif command[0]==7:
        if len(d)==0: print(-1)
        else: print(d[0])
    elif command[0]==8:
        if len(d)==0: print(-1)
        else: print(d[-1])
```
