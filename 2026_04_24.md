### python
```python
n=int(input())
point = []
for i in range(n):
    point.append(tuple(map(int,input().split())))
l=0
for i in range(n-1):
    if point[i][0]==point[i+1][0]: l+=abs(point[i][1]-point[i+1][1])
    elif point[i][1]==point[i+1][1]: l+=abs(point[i][0]-point[i+1][0])
if point[-1][0]==point[0][0]: l+=abs(point[-1][1]-point[0][1])
elif point[-1][1]==point[0][1]: l+=abs(point[-1][0]-point[0][0])
print(l)
```
- 입력받는 순서대로 연결하는 조건이라 상대적으로 간단
- 반드시 직각으로 이어지므로 x좌표가 같은지 y좌표가 같은지 판단 후 길이 측정

### java
```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

class Point{
    int x;
    int y;
    Point(int x, int y){
        this.x = x;
        this.y = y;
    }
}

public class p15923 {
    public static void main(String[] args)throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int n=Integer.parseInt(br.readLine());
        Point[] point = new Point[n];
        for (int i=0;i<n;i++) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            int x = Integer.parseInt(st.nextToken());
            int y = Integer.parseInt(st.nextToken());
            point[i] = new Point(x,y);
        }
        int l = 0;
        for (int i=0;i<n-1;i++){
            if (point[i].x == point[i+1].x) l+=Math.abs(point[i].y-point[i+1].y);
            else if (point[i].y == point[i+1].y) l+=Math.abs(point[i].x-point[i+1].x);
        }
        if (point[0].x == point[n-1].x) l+=Math.abs(point[0].y-point[n-1].y);
        else if (point[0].y == point[n-1].y) l+=Math.abs(point[0].x-point[n-1].x);
        System.out.println(l);
    }
}
```
