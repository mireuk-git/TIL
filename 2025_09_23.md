```python
count=0
stop=False
def heapify(A,k,n):
    global count,stop
    if stop:
        return
    left,right=2*k,2*k+1
    if right<=n:
        if A[left]<A[right]: smaller = left
        else: smaller = right
    elif (left<=n): smaller=left
    else: return

    if A[smaller]<A[k]:
        A[k],A[smaller]=A[smaller],A[k]
        count+=1
        if count == k_limit:
            stop=True
            return
        heapify(A,smaller,n)

def build_min_heap(A,n):
    for i in range(n//2,0,-1):
        heapify(A,i,n)

def heap_sort(A):
    global count, stop
    build_min_heap(A,a)
    for i in range(a,1,-1):
        if stop: break
        A[1],A[i] = A[i],A[1]
        count+=1
        if count==k_limit:
            stop=True
            break
        heapify(A,1,i-1)

a,k_limit=map(int,input().split())
A=[0]+list(map(int,input().split()))
heap_sort(A)
if count<k_limit: print(-1)
else:
    for i in range(1,a+1): print(A[i],end=" ")
```
