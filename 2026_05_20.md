### python
```python
def solution(board, h, w):
    n=len(board)
    count=0
    dh,dw = [0,1,-1,0],[1,0,0,-1]
    for k in range(4):
        h_ch,w_ch=h+dh[k],w+dw[k]
        if 0<=h_ch<n and 0<=w_ch<n :
            if board[h][w]==board[h_ch][w_ch]:
                count+=1
    return count
```

### java
```java
class Solution{
    public int solution(String[][]board, int h, int w){
        int n = board.length;
        int count=0;
        int[] dh={0,1,-1,0}, dw={1,0,0,-1};
        for(int i=0;i<4;i++){
            int hCh=h+dh[i], wCh=w+dw[i];
            if (0<=hCh && hCh<n && 0<=wCh && wCh<n)
            if (board[h][w].equals(board[hCh][wCh])) count++;
        }
        return count;
    }
}
```
